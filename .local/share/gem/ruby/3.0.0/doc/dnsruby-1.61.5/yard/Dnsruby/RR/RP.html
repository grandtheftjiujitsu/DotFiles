def from_string(input) #:nodoc: all
  if (input.length > 0)
    names = input.split(" ")
    @mailbox = Name.create(names[0])
    @txtdomain = Name.create(names[1])
  end
end

def rdata_to_string #:nodoc: all
  if (@mailbox!=nil)
    return "#{@mailbox.to_s(true)} #{@txtdomain.to_s(true)}"
  else
    return ""
  end
end
