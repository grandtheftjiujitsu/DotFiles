def from_string(input) #:nodoc: all
  if (input.length > 0)
    names = input.split(" ")
    @preference = names[0].to_i
    @intermediate = Name.create(names[1])
  end
end

def rdata_to_string #:nodoc: all
  if (@preference!=nil)
    return "#{@preference} #{@intermediate.to_s(true)}"
  else
    return ""
  end
end
