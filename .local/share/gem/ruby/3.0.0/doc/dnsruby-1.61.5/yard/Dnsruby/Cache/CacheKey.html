def initialize(*args)
  self.qclass = Classes.IN
  if (args.length > 0)
    self.qname = Name.create(args[0])
    self.qname.absolute = true
    if (args.length > 1)
      self.qtype = Types.new(args[1])
      if (args.length > 2)
        self.qclass = Classes.new(args[2])
      end
    end
  end
end
