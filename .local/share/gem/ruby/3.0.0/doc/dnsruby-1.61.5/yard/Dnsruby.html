def log_and_raise(object, error_class = RuntimeError)
  if object.is_a?(Exception)
    error = object
    Dnsruby.log.error(error.inspect)
    raise error
  else
    message = object.to_s
    Dnsruby.log.error(message)
    raise error_class.new(message)
  end
end
